// Function signature and definition of the equational theory E

functions: fst/1, h/1, pair/2, snd/1
equations: fst(<x.1, x.2>) = x.1, snd(<x.1, x.2>) = x.2

rule (modulo E) OracleSetup:
   [ Fr( ~oracle_seed ) ]
  --[ OracleInitialized( h(~oracle_seed) ) ]->
   [
   !OracleState( ~oracle_seed ), !OraclePublicRoot( h(~oracle_seed) ),
   Out( h(~oracle_seed) )
   ]

  /* has exactly the trivial AC variant */

rule (modulo E) IssuePair:
   [
   !OracleState( ~oracle_seed ), Fr( ~iss ), Fr( ~credA ), Fr( ~credB ),
   Fr( ~window ), Fr( ~space )
   ]
  --[
  OracleIssued( ~iss,
                h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                   ~space>),
                $A, $B, ~window, ~space
  ),
  PairIssued( ~iss )
  ]->
   [
   !OracleRoot( ~iss,
                h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                   ~space>),
                $A, $B, ~window, ~space
   ),
   PairOpen( ~iss, $A, $B, ~credA, ~credB, ~window, ~space,
             h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                ~space>)
   ),
   ParentACred( $A, ~iss, ~credA,
                h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                   ~space>)
   ),
   ParentBCred( $B, ~iss, ~credB,
                h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                   ~space>)
   ),
   Out( <~iss, 
         h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
            ~space>), 
         $A, $B, ~window, ~space>
   )
   ]

  /* has exactly the trivial AC variant */

rule (modulo E) ParentAParticipate:
   [ ParentACred( A, iss, credA, root ), Fr( ~timeA ), Fr( ~contextA ) ]
  --[ ParentAParticipated( iss, A, root, ~timeA, ~contextA ) ]->
   [ ParentAReady( A, iss, credA, root, ~timeA, ~contextA ) ]

  /* has exactly the trivial AC variant */

rule (modulo E) ParentBParticipate:
   [ ParentBCred( B, iss, credB, root ), Fr( ~timeB ), Fr( ~contextB ) ]
  --[ ParentBParticipated( iss, B, root, ~timeB, ~contextB ) ]->
   [ ParentBReady( B, iss, credB, root, ~timeB, ~contextB ) ]

  /* has exactly the trivial AC variant */

rule (modulo E) GenerateChild:
   [
   PairOpen( iss, A, B, credA, credB, window, space, root ),
   ParentAReady( A, iss, credA, root, timeA, contextA ),
   ParentBReady( B, iss, credB, root, timeB, contextB ), Fr( ~timeC )
   ]
  --[
  PairConsumed( iss ),
  ChildGenerated( iss,
                  h(<'oclm-child-v1', iss, root, A, B, credA, credB, timeA, timeB, ~timeC, 
                     window, space, contextA, contextB>),
                  root, A, B, timeA, timeB, ~timeC, window, space
  )
  ]->
   [
   !ChildRecord( iss,
                 h(<'oclm-child-v1', iss, root, A, B, credA, credB, timeA, timeB, ~timeC, 
                    window, space, contextA, contextB>),
                 root, A, B, timeA, timeB, ~timeC, window, space
   ),
   Out( <iss, 
         h(<'oclm-child-v1', iss, root, A, B, credA, credB, timeA, timeB, ~timeC, 
            window, space, contextA, contextB>), 
         root, A, B, ~timeC, window, space>
   )
   ]

  /* has exactly the trivial AC variant */

rule (modulo E) VerifyChild:
   [
   !ChildRecord( iss, child, root, A, B, timeA, timeB, timeC, window, space
   ),
   !OracleRoot( iss, root, A, B, window, space )
   ]
  --[
  ChildAccepted( iss, child, root, A, B, timeA, timeB, timeC, window, space
  )
  ]->
   [ ]

  /* has exactly the trivial AC variant */

rule (modulo E) PublishClaim:
   [ In( <iss, claim> ) ] --[ ClaimPublished( iss, claim ) ]-> [ ]

  /* has exactly the trivial AC variant */

rule (modulo E) KeyCompromise:
   [ !OracleState( oracle_seed ) ]
  --[ OracleKeyCompromised( h(oracle_seed) ) ]->
   [ Out( oracle_seed ) ]

  /* has exactly the trivial AC variant */

rule (modulo E) OracleEquivocation:
   [ !OracleRoot( iss, root1, A, B, window, space ), Fr( ~root2 ) ]
  --[
  OracleEquivocated( iss, root1, ~root2 ),
  EquivocationEvidence( iss, root1, ~root2 )
  ]->
   [ Out( <iss, root1, ~root2> ) ]

  /* has exactly the trivial AC variant */

lemma executable:
  exists-trace
  "∃ iss child root A B timeA timeB timeC window space #i.
    ChildAccepted( iss, child, root, A, B, timeA, timeB, timeC, window, space
    ) @ #i"
/*
guarded formula characterizing all satisfying traces:
"∃ iss child root A B timeA timeB timeC window space #i.
  (ChildAccepted( iss, child, root, A, B, timeA, timeB, timeC, window,

   ) @ #i)"

  qed

         ) ▶₁ #i )

    solve( ParentBReady( $B, ~iss, ~credB,
                         h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                            ~space>),

      case ParentBParticipate
      by contradiction /* from formulas */

  qed

lemma complementary_pair_is_single_use:
  all-traces
  "∀ iss child1 root1 A1 B1 timeA1 timeB1 timeC1 window1 space1 child2
     root2 A2 B2 timeA2 timeB2 timeC2 window2 space2 #i #j.
    ((ChildGenerated( iss, child1, root1, A1, B1, timeA1, timeB1, timeC1,
                      window1, space1
      ) @ #i) ∧
     (ChildGenerated( iss, child2, root2, A2, B2, timeA2, timeB2, timeC2,
                      window2, space2
      ) @ #j)) ⇒
    (#i = #j)"
/*
guarded formula characterizing all counter-examples:
"∃ iss child1 root1 A1 B1 timeA1 timeB1 timeC1 window1 space1 child2
   root2 A2 B2 timeA2 timeB2 timeC2 window2 space2 #i #j.
  (ChildGenerated( iss, child1, root1, A1, B1, timeA1, timeB1, timeC1,
                   window1, space1
   ) @ #i) ∧
  (ChildGenerated( iss, child2, root2, A2, B2, timeA2, timeB2, timeC2,
                   window2, space2
   ) @ #j)
 ∧
  ¬(#i = #j)"
*/
simplify
solve( (#i < #j)  ∥ (#j < #i) )
  case case_1
  solve( PairOpen( iss, A1, B1, credA, credB, window1, space1, root1
         ) ▶₀ #i )
    case IssuePair
    solve( ParentAReady( $A, ~iss, ~credA,
                         h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                            ~space>),
                         timeA1, contextA
           ) ▶₁ #i )
      case ParentAParticipate
      solve( ParentBReady( $B, ~iss, ~credB,
                           h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window, 
                              ~space>),
                           timeB1, contextB
             ) ▶₂ #i )
        case ParentBParticipate
        solve( PairOpen( ~iss, A2, B2, credA.1, credB.1, window2, space2, root2
               ) ▶₀ #j )
          case IssuePair
          by contradiction /* cyclic */
        qed
      qed
    qed
  qed
next
  case case_2
  solve( PairOpen( iss, A1, B1, credA, credB, window1, space1, root1
         ) ▶₀ #i )
    case IssuePair
    solve( ParentAReady( $A, ~iss, ~credA,
                         h(<'oclm-root-v1', ~oracle_seed, ~iss, $A, $B, ~credA, ~credB, ~window,

      solve( ParentBReady( $B, ~iss, ~credB,

        solve( PairOpen( ~iss, A2, B2, credA.1, credB.1, window2, space2, root2
